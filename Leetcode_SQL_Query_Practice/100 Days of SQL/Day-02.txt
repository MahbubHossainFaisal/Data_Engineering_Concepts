-- find employees with salary more than their manager's salary

with cte as 
(
select distinct emp.emp_id,emp.emp_name,emp.salary,emp.manager_id,mgr.emp_name as manager_name,mgr.salary
from emp emp
join emp mgr
on emp.manager_id = mgr.emp_id
where mgr.salary < emp.salary
)
select emp_name from cte order by emp_name;
