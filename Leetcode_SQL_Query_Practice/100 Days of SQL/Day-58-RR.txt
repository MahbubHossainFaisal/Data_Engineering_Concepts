/*
For each region, find the house which has won the maximum number of battles. Display region, house, and number of wins.
*/

-- Create the 'king' table
CREATE TABLE king (
    k_no INT PRIMARY KEY,
    king VARCHAR(50),
    house VARCHAR(50)
);

-- Create the 'battle' table
CREATE TABLE battle (
    battle_number INT PRIMARY KEY,
    name VARCHAR(100),
    attacker_king INT,
    defender_king INT,
    attacker_outcome INT,
    region VARCHAR(50),
    FOREIGN KEY (attacker_king) REFERENCES king(k_no),
    FOREIGN KEY (defender_king) REFERENCES king(k_no)
);

with winners as
(
select attacker_king as winning_king, region
from battle where attacker_outcome = 1 
union all
select defender_king as winning_king, region
from battle where attacker_outcome = 0
),
winner_with_house as
(
select winning_king,region,house from 
winners a 
left join king b
on a.winning_king = b.k_no
),
agg as 
(
select region, house, count(*) as no_of_wins,dense_rank() over(partition by region order by count(*) desc) as rn
from winner_with_house
group by region,house
order by region,house
)
select * 
from agg where rn=1
order by house desc;
