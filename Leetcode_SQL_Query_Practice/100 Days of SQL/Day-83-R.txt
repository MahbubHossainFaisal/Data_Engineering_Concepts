with cte as
(
select team_1,team_2, case when team_1=winner then 1 else 0 end as team1_winning_flag,
case when team_2=winner then 1 else 0 end as team2_winning_flag from icc_world_cup
where winner <> 'DRAW'
)
select team, count(*) as matches_played, sum(winning_status) as wins, count(case when winning_status = 0 then 1 end) as losses, sum(winning_status)*2 as pts
from 
(select team_1 as team, team1_winning_flag as winning_status from cte
union all 
select team_2, team2_winning_flag from cte) A
group by team
order by wins desc;
