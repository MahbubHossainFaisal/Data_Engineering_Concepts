-- Method one subquery

with cte as
(
select gold as player_Name from events
where gold not in (select silver from events) 
and gold not in (select bronze from events)
)
select player_Name,count(*) as no_of_medals
from cte
group by player_Name
order by player_Name;


-- Method 2 
with cte as
(
select gold as player_Name from events
where gold not in (select silver from events union all select bronze from events) 
)
select player_Name,count(*) as no_of_medals
from cte
group by player_Name
order by player_Name;


-- Method 3

WITH cte2 AS (
    SELECT UPPER(gold) AS player_Name, 'gold' AS medal_type FROM events
    UNION ALL
    SELECT UPPER(silver), 'silver' AS medal_type FROM events
    UNION ALL 
    SELECT UPPER(bronze), 'bronze' AS medal_type FROM events
)
SELECT player_Name, COUNT(*) AS no_of_medals
FROM cte2
GROUP BY player_Name
HAVING COUNT(DISTINCT medal_type) = 1 and max(medal_type)='gold';
